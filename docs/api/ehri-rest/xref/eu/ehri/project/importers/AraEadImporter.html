43  /**
44   * Import EAD for a given repository into the database. Due to the laxness of the EAD standard this is a fairly complex
45   * procedure. An EAD a single entity at the highest level of description or multiple top-level entities, with or without
46   * a hierarchical structure describing their child items. This means that we need to recursively descend through the
47   * archdesc and c,c01-12 levels.
48   * <p>
49   * will preserve existing 'otherIdentifiers' on the DocumentaryUnit.
50   * <p>
51   * Furthermore, it will always try to resolve the UndeterminedRelationships, not just on creation.
52   * This is not standard behaviour, so use with caution.
53   * <p>
54   * TODO: Extensive cleanups, optimisation, and rationalisation.
55   *
56   * @author Linda Reijnhoudt (https://github.com/lindareijnhoudt)
57   */
58  public class AraEadImporter extends EadImporter {
59  
60      private static final Logger logger = LoggerFactory.getLogger(AraEadImporter.class);
61      //the EadImporter can import ead as DocumentaryUnits, the default, or overwrite those and create VirtualUnits instead.
62      private final EntityClass unitEntity = EntityClass.DOCUMENTARY_UNIT;
63      private final Serializer mergeSerializer;
64  
65      /**
66       * Construct an EadImporter object.
67       *
68       * @param graph           the framed graph
69       * @param permissionScope the permission scope
70       * @param log             the import log
71       */
72      public AraEadImporter(FramedGraph<?> graph, PermissionScope permissionScope, ImportLog log) {
73          super(graph, permissionScope, log);
74          mergeSerializer = new Serializer.Builder(graph).dependentOnly().build();
75      }

77      /**
78       * Import a single archdesc or c01-12 item, keeping a reference to the hierarchical depth.
79       *
80       * @param itemData The data map
81       * @param idPath   The identifiers of parent documents, not including those of the overall permission scope
82       * @throws ValidationError when the itemData does not contain an identifier for the unit or...
83       */
84      @Override
85      public AbstractUnit importItem(Map<String, Object> itemData, List<String> idPath)
86              throws ValidationError {
87  
88          BundleDAO persister = getPersister(idPath);
89  
90          List<Map<String, Object>> extractedDates = extractDates(itemData);
91          replaceDates(itemData, extractedDates);
92  
93          Bundle descBundle = new Bundle(EntityClass.DOCUMENT_DESCRIPTION, extractUnitDescription(itemData, EntityClass.DOCUMENT_DESCRIPTION));
94          // Add dates and descriptions to the bundle since they're @Dependent
95          // relations.
96          for (Map<String, Object> dpb : extractedDates) {
97              descBundle = descBundle.withRelation(Ontology.ENTITY_HAS_DATE, new Bundle(EntityClass.DATE_PERIOD, dpb));
98          }
99          for (Map<String, Object> rel : extractRelations(itemData)) {//, (String) unit.getErrors().get(IdentifiableEntity.IDENTIFIER_KEY)
100             logger.debug("relation found: " + rel.get(Ontology.NAME_KEY));
101             for (String s : rel.keySet()) {
102                 logger.debug(s);
103             }
104             descBundle = descBundle.withRelation(Ontology.HAS_ACCESS_POINT, new Bundle(EntityClass.UNDETERMINED_RELATIONSHIP, rel));
105         }
106         Map<String, Object> unknowns = extractUnknownProperties(itemData);
107         if (!unknowns.isEmpty()) {
108             StringBuilder unknownProperties = new StringBuilder();
109             for (String u : unknowns.keySet()) {
110                 unknownProperties.append(u);
111             }
112             logger.info("Unknown Properties found: " + unknownProperties);
113             descBundle = descBundle.withRelation(Ontology.HAS_UNKNOWN_PROPERTY, new Bundle(EntityClass.UNKNOWN_PROPERTY, unknowns));
114         }
115         // extractDocumentaryUnit does not throw ValidationError on missing ID
116         Bundle unit = new Bundle(unitEntity, extractDocumentaryUnit(itemData));
117 
118 
119         // Check for missing identifier, throw an exception when there is no ID.
120         if (unit.getDataValue(Ontology.IDENTIFIER_KEY) == null) {
121             throw new ValidationError(unit, Ontology.IDENTIFIER_KEY,
122                     "Missing identifier " + Ontology.IDENTIFIER_KEY);
123         }
124         logger.debug("Imported item: " + itemData.get("name"));
125 
126         Mutation<DocumentaryUnit> mutation =
127                 persister.createOrUpdate(mergeWithPreviousAndSave(unit, descBundle, idPath), DocumentaryUnit.class);
128         DocumentaryUnit frame = mutation.getNode();
129 
130         // Set the repository/item relationship
131         if (idPath.isEmpty() && mutation.created()) {
132             EntityClass scopeType = manager.getEntityClass(permissionScope);
133             if (scopeType.equals(EntityClass.REPOSITORY)) {
134                 Repository repository = framedGraph.frame(permissionScope.asVertex(), Repository.class);
135                 frame.setRepository(repository);
136                 frame.setPermissionScope(repository);
137             } else if (scopeType.equals(unitEntity)) {
138                 DocumentaryUnit parent = framedGraph.frame(permissionScope.asVertex(), DocumentaryUnit.class);
139                 parent.addChild(frame);
140                 frame.setPermissionScope(parent);
141             } else {
142                 logger.error("Unknown scope type for documentary unit: {}", scopeType);
143             }
144         }
145         handleCallbacks(mutation);
146         logger.debug("============== " + frame.getIdentifier() + " created:" + mutation.created());
147 
148         //BEWARE: it will always try to solve the UndeterminedRelationships, not only on creation!
149 //        if (mutation.created()) {
150         solveUndeterminedRelationships(frame, descBundle);
151 //        }
152         return frame;
153 
154 
155     }

157     /**
158      * finds any bundle in the graph with the same ObjectIdentifier. if it exists it replaces the Description in the
159      * given language, else it just saves it
160      *
161      * @param unit       - the DocumentaryUnit to be saved
162      * @param descBundle - the documentsDescription to replace any previous ones with this language
163      * @return A bundle with description relationships merged.
164      * @throws ValidationError
165      */
166     protected Bundle mergeWithPreviousAndSave(Bundle unit, Bundle descBundle, List<String> idPath) throws ValidationError {
167         final String languageOfDesc = descBundle.getDataValue(Ontology.LANGUAGE_OF_DESCRIPTION);
168         final String thisSourceFileId = descBundle.getDataValue(Ontology.SOURCEFILE_KEY);
169         /*
170          * for some reason, the idpath from the permissionscope does not contain the parent documentary unit.
171          * TODO: so for now, it is added manually
172          */
173         List<String> lpath = Lists.newArrayList();
174         for (String p : getPermissionScope().idPath()) {
175             lpath.add(p);
176         }
177         for (String p : idPath) {
178             lpath.add(p);
179         }
180         Bundle withIds = unit.generateIds(lpath);
181 
182         if (manager.exists(withIds.getId())) {
183             try {
184                 //read the current item’s bundle
185                 Bundle oldBundle = mergeSerializer
186                         .vertexFrameToBundle(manager.getVertex(withIds.getId()));
187 
188                 //determine if previous existing DocUnit had 'otherIdentifiers', if so, add to existing withIds
189                 if (oldBundle.getData().keySet().contains(Ontology.OTHER_IDENTIFIERS)) {
190                     Object otherIdentifiers = oldBundle.getData().get(Ontology.OTHER_IDENTIFIERS);
191                     if (unit.getData().keySet().contains(Ontology.OTHER_IDENTIFIERS)) {
192                         if (otherIdentifiers instanceof List) {
193                             ((List<String>) otherIdentifiers).add(unit.getDataValue(Ontology.OTHER_IDENTIFIERS).toString());
194                         } else if (otherIdentifiers instanceof String) {
195                             List<String> allOtherIdentifiers = Lists.newArrayList();
196                             allOtherIdentifiers.add(otherIdentifiers.toString());
197                             allOtherIdentifiers.add(unit.getDataValue(Ontology.OTHER_IDENTIFIERS).toString());
198                             otherIdentifiers = allOtherIdentifiers;
199                         }
200                     }
201                     withIds = withIds.withDataValue(Ontology.OTHER_IDENTIFIERS, otherIdentifiers);
202                 }
203 
204 
205                 //if the unit exists, with a desc with the same sourcefileid, overwrite, else create new desc
206                 //filter out dependents that a) are descriptions, b) have the same language/code
207                 Bundle.Filter filter = new Bundle.Filter() {
208                     @Override
209                     public boolean remove(String relationLabel, Bundle bundle) {
210                         String lang = bundle.getDataValue(Ontology.LANGUAGE);
211                         String oldSourceFileId = bundle.getDataValue(Ontology.SOURCEFILE_KEY);
212                         return bundle.getType().equals(EntityClass.DOCUMENT_DESCRIPTION)
213                                 && (lang != null
214                                 && lang.equals(languageOfDesc)
215                                 && (oldSourceFileId != null && oldSourceFileId.equals(thisSourceFileId)));
216                     }
217                 };
218                 Bundle filtered = oldBundle.filterRelations(filter);
219 
220                 //if this desc-id already exists, but with a different sourceFileId, 
221                 //change the desc-id
222                 String defaultDescIdentifier = withIds.getId() + "-" + languageOfDesc.toLowerCase();
223                 String newDescIdentifier = withIds.getId() + "-" + thisSourceFileId.toLowerCase().replace("#", "-");
224                 if (manager.exists(newDescIdentifier)) {
225                     descBundle = descBundle.withDataValue(Ontology.IDENTIFIER_KEY, newDescIdentifier);
226                 } else if (manager.exists(defaultDescIdentifier)) {
227                     Bundle oldDescBundle = mergeSerializer
228                             .vertexFrameToBundle(manager.getVertex(defaultDescIdentifier));
229                     //if the previous had NO sourcefile_key OR it was different:
230                     if (oldDescBundle.getDataValue(Ontology.SOURCEFILE_KEY) == null
231                             || !thisSourceFileId.equals(oldDescBundle.getDataValue(Ontology.SOURCEFILE_KEY).toString())) {
232                         descBundle = descBundle.withDataValue(Ontology.IDENTIFIER_KEY, newDescIdentifier);
233                         logger.info("other description found (" + defaultDescIdentifier + "), creating new description id: " + descBundle.getDataValue(Ontology.IDENTIFIER_KEY));
234                     }
235                 }
236 
237                 return withIds.withRelations(filtered.getRelations())
238                         .withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
239 
240             } catch (SerializationError ex) {
241                 throw new ValidationError(unit, "serialization error", ex.getMessage());
242             } catch (ItemNotFound ex) {
243                 throw new ValidationError(unit, "item not found exception", ex.getMessage());
244             }
245         } else {
246             return unit.withRelation(Ontology.DESCRIPTION_FOR_ENTITY, descBundle);
247         }
248     }
