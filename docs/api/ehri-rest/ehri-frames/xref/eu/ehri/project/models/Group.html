43  /**
44   * Frame class representing a group of users or other groups
45   * that can be assigned permissions.
46   *
47   * @author Mike Bryant (http://github.com/mikesname)
48   */
49  @EntityType(EntityClass.GROUP)
50  public interface Group extends Accessor, AccessibleEntity, IdentifiableEntity,
51          PermissionScope, NamedEntity, ItemHolder {
52      
53      public static final String ADMIN_GROUP_IDENTIFIER = "admin";
54      public static final String ANONYMOUS_GROUP_IDENTIFIER = "anonymous";
55      public static final String ADMIN_GROUP_NAME = "Administrators";
56  
57      /**
58       * Fetch the groups to which this group belongs.
59       *
60       * @return an iterable of group frames
61       */
62      @Fetch(Ontology.ACCESSOR_BELONGS_TO_GROUP)
63      @Adjacency(label = Ontology.ACCESSOR_BELONGS_TO_GROUP)
64      public Iterable<Group> getGroups();
65  
66      /**
67       * TODO FIXME use this in case we need AccesibleEnity's instead of Accessors, 
68       */
69      @Adjacency(label = Ontology.ACCESSOR_BELONGS_TO_GROUP, direction = Direction.IN)
70      public Iterable<AccessibleEntity> getMembersAsEntities();
71  
72      /**
73       * Get members of this group.
74       *
75       * @return an iterable of user or group frames
76       */
77      @Adjacency(label = Ontology.ACCESSOR_BELONGS_TO_GROUP, direction = Direction.IN)
78      public Iterable<Accessor> getMembers();
79  
80      /**
81       * Get the number of items within this group.
82       *
83       * @return a count of group members
84       */
85      @Meta(CHILD_COUNT)
86      @JavaHandler
87      public long getChildCount();
88  
89      /**
90       * Adds a Accessor as a member to this Group, so it has the permissions of the Group.
91       *
92       * @param accessor a user or group frame
93       */
94      @JavaHandler
95      public void addMember(final Accessor accessor);
96  
97      /**
98       * Removes a member from this group.
99       *
100      * @param accessor a user or group frame
101      */
102     @JavaHandler
103     public void removeMember(final Accessor accessor);
104 
105     /**
106      * Get <b>all</b> members of this group, including members of
107      * groups within this group.
108      *
109      * @return an iterable of user or group frames
110      */
111     @JavaHandler
112     public Iterable<AccessibleEntity> getAllUserProfileMembers();
113 
114     /**
115      * Implementation of complex methods.
116      */
117     abstract class Impl implements JavaHandlerContext<Vertex>, Group {
118 
119         public long getChildCount() {
120             return gremlin().inE(Ontology.ACCESSOR_BELONGS_TO_GROUP).count();
121         }
122 
123         public void addMember(final Accessor accessor) {
124             JavaHandlerUtils.addUniqueRelationship(accessor.asVertex(), it(),
125                     Ontology.ACCESSOR_BELONGS_TO_GROUP);
126         }
127 
128         public void removeMember(final Accessor accessor) {
129             JavaHandlerUtils.removeAllRelationships(accessor.asVertex(),
130                     it(), Ontology.ACCESSOR_BELONGS_TO_GROUP);
131         }
132 
133         public Iterable<AccessibleEntity> getAllUserProfileMembers() {
134             GremlinPipeline<Vertex,Vertex> pipe = gremlin().as("n").in(Ontology.ACCESSOR_BELONGS_TO_GROUP)
135                     .loop("n", JavaHandlerUtils.defaultMaxLoops, new PipeFunction<LoopPipe.LoopBundle<Vertex>, Boolean>() {
136                         @Override
137                         public Boolean compute(LoopPipe.LoopBundle<Vertex> vertexLoopBundle) {
138                             return vertexLoopBundle.getObject()
139                                     .getProperty(EntityType.TYPE_KEY)
140                                     .equals(Entities.USER_PROFILE);
141                         }
142                     });
143             return frameVertices(pipe.dedup());
144         }
145     }
